1. Escreva uma função em C que faz o debounce de botões ligados à porta P1.

#include <msp430g2553.h>
#define BTN  BIT3

int main(void)
{
	WDTCTL = WDTPW | WDTHOLD;
	P1OUT |= LEDS;
	P1DIR |= LEDS;
	P1DIR &= ~BTN;
	P1REN |= BTN;
	P1OUT |= BTN;
	while(1)
	{
		if((P1IN&BTN)==0)
			BTN = debounce();
		else
			P1OUT &= ~LEDS;
	}
	return 0;
}

int debounce()
{
  int i = 0;
  while(i < 100)
  {
    if((P1IN&BTN)==0)
    {
      i++;
      return 1;
    }
    else
      return 0;

  } 
}
