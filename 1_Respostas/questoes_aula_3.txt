3.Considerando a placa Launchpad do MSP430, escreva o código em C para piscar duas vezes os dois 
LEDs sempre que o usuário pressionar o botão.
#include <msp430g2553.h>
#define LED1 BIT0
#define LED2 BIT6
#define BTN BIT2

void main (void)
{
  WDTCTL = WDTPW | WDTHOLD;
  P1DIR = BIT6 + BIT0;
  P1OUT = BIT6 + BIT0;
  
  for (;;)
  {
    if (P1IN & BTN == 0)
    P1DIR ^= BIT6 + BIT0;
  }
}
